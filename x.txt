DROP TABLE IF EXISTS pivo;
DROP TABLE IF EXISTS slad;
DROP TABLE IF EXISTS chmel;
DROP TABLE IF EXISTS kvasnice;
DROP TABLE IF EXISTS zaznam_vypiteho_piva;
DROP TABLE IF EXISTS hodnoceni;
DROP TABLE IF EXISTS uzivatel;
DROP TABLE IF EXISTS hospoda;
DROP TABLE IF EXISTS distribuce_piva;
DROP TABLE IF EXISTS kamenna_prodejna;
DROP TABLE IF EXISTS ramcova_smlouva;
DROP TABLE IF EXISTS pivovar;
DROP TABLE IF EXISTS doba_vareni;
DROP TABLE IF EXISTS sladek;
DROP TABLE IF EXISTS amatersky_sladek;
DROP TABLE IF EXISTS konzument;
CREATE TABLE pivo(ID_pivo INT NOT NULL AUTO_INCREMENT,nazev VARCHAR(25) NOT NULL,stupen_EBC INT NOT NULL,styl_kvaseni VARCHAR(25) NOT NULL,typ_piva VARCHAR(25) NOT NULL,obsah_alkoholu INT NOT NULL, PRIMARY KEY (ID_pivo));
CREATE TABLE slad(ID_slad INT NOT NULL AUTO_INCREMENT,barva VARCHAR(25) NOT NULL,puvod VARCHAR(25) NOT NULL,extrakt VARCHAR(25) NOT NULL, PRIMARY KEY (ID_slad));
CREATE TABLE chmel(ID_chmel INT NOT NULL AUTO_INCREMENT,puvod VARCHAR(25) NOT NULL,doba_sklizne VARCHAR(25) NOT NULL,aroma VARCHAR(25) NOT NULL,horkost INT NOT NULL,podil_alfa_kyselin INT NOT NULL, PRIMARY KEY (ID_chmel));
CREATE TABLE kvasnice(ID_kvasnice INT NOT NULL AUTO_INCREMENT,skupenstvi VARCHAR(25) NOT NULL,typ_kvaseni VARCHAR(25) NOT NULL, PRIMARY KEY(ID_kvasnice));
CREATE TABLE zaznam_vypiteho_piva(login VARCHAR(25) NOT NULL,ID_pivo INT NOT NULL,datum DATE NOT NULL,cas VARCHAR(25) NOT NULL, PRIMARY KEY (login,ID_pivo));
CREATE TABLE doba_vareni(ID_pivo INT NOT NULL,ID_sladka INT NOT NULL,od timestamp NOT NULL,do timestamp NOT NULL, PRIMARY KEY (ID_pivo,ID_sladka));
CREATE TABLE hodnoceni(ID_pivo INT NOT NULL,login VARCHAR(25) NOT NULL,ID_hospoda INT NOT NULL,znamka INT NOT NULL,datum DATE NOT NULL,cas VARCHAR(25) NOT NULL, PRIMARY KEY (ID_pivo,login,ID_hospoda));
CREATE TABLE hospoda(ID_hospoda INT NOT NULL AUTO_INCREMENT,nazev VARCHAR(25) NOT NULL,adr_mesto VARCHAR(30) NOT NULL,adr_ulice VARCHAR(30) NOT NULL,adr_cislo_domu INT NOT NULL,PSC INT NOT NULL, PRIMARY KEY (ID_hospoda));
CREATE TABLE distribuce_piva(ID_pivo INT NOT NULL,ID_hospoda INT NOT NULL,ID_prodejna INT NOT NULL,nadoba INT NOT NULL, mnozstvi INT NOT NULL, PRIMARY KEY (ID_pivo,ID_hospoda,ID_prodejna));
CREATE TABLE kamenna_prodejna(ID_prodejna INT NOT NULL AUTO_INCREMENT,nazev VARCHAR(25) NOT NULL,adr_mesto VARCHAR(30) NOT NULL,adr_ulice VARCHAR(30) NOT NULL,adr_cislo_domu INT NOT NULL,PSC INT (5) NOT NULL, PRIMARY KEY (ID_prodejna));
CREATE TABLE amatersky_sladek(login VARCHAR(25) NOT NULL,mnozstvi_uvareneho_piva INT NOT NULL,celkova_cena_surovin INT NOT NULL, PRIMARY KEY (login));
CREATE TABLE konzument(login VARCHAR(25) NOT NULL,mesicni_rozpocet INT NOT NULL, PRIMARY KEY (login));
CREATE TABLE ramcova_smlouva(ID_pivovar INT NOT NULL,ID_hospoda INT NOT NULL,platnost_od DATE NOT NULL,platnost_do DATE NOT NULL,sleva INT NOT NULL, PRIMARY KEY (ID_pivovar,ID_hospoda));
CREATE TABLE pivovar(ID_pivovar INT NOT NULL AUTO_INCREMENT,nazev VARCHAR(25) NOT NULL,adr_mesto VARCHAR(30) NOT NULL,adr_ulice VARCHAR(30) NOT NULL,adr_cislo_domu INT NOT NULL,PSC INT (5) NOT NULL, PRIMARY KEY (ID_pivovar));
CREATE TABLE uzivatel(login VARCHAR(25) NOT NULL,heslo VARCHAR(25) NOT NULL, PRIMARY KEY (login));
CREATE TABLE sladek(ID_sladka INT NOT NULL AUTO_INCREMENT,login VARCHAR(25) NOT NULL,jmeno VARCHAR(25) NOT NULL,prijmeni VARCHAR(25) NOT NULL,adr_mesto VARCHAR(30) NOT NULL,adr_ulice VARCHAR(30) NOT NULL,adr_cislo_domu INT NOT NULL,PSC INT (5) NOT NULL, PRIMARY KEY (ID_sladka));
